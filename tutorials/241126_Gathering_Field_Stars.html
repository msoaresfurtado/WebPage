def get_gaia_dr3_params():
    """
    Query Gaia DR3 and return the results in a pandas DataFrame.
    """
    query = f"""
    SELECT TOP 25000
        SOURCE_ID, ra, dec, pmra, pmdec
    FROM gaiadr3.gaia_source
    WHERE parallax > 10 AND RUWE < 1.2 AND parallax_over_error > 5 
    AND phot_g_mean_flux_over_error>50. AND phot_rp_mean_flux_over_error>20
    """
    # Execute the query using Gaia API
    job = Gaia.launch_job_async(query)
    results = job.get_results()
    # Check if results are returned
    if len(results) == 0:
        print("No data found matching the query.")
        return None
    
    # Convert results to pandas DataFrame
    df = results.to_pandas()
    
    # Display the DataFrame
    print(df)
    
    return df

# Retrieve data and store it in a DataFrame
gaia_data = get_gaia_dr3_params()
