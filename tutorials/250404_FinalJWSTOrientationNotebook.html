import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from astropy.wcs import WCS
from astropy.io import fits
from astropy.visualization import ImageNormalize, AsinhStretch
from astropy import units as u
from astropy.coordinates import SkyCoord
from astroquery.gaia import Gaia
from matplotlib.patches import Rectangle
from matplotlib.path import Path
from matplotlib.transforms import Affine2D

fits_file = '/Users/melindasoares/Desktop/jwst_ONC/MAST_2025-03-21T1906/JWST/jw01256-o001_t001_nircam_f150w2-f162m/rotated_ds9.fits'
index_choice = 0
with fits.open(fits_file, mode="readonly") as hdulist:
    wcs_info = WCS(hdulist[index_choice].header)
    cal_image = hdulist[index_choice].data
    header = hdulist[index_choice].header

infile1 = '/Users/melindasoares/Desktop/jwst_ONC/robberto_datafile5_rev.csv'
infile2 = '/Users/melindasoares/Desktop/jwst_ONC/robberto_datafile2_rev.csv'
df1 = pd.read_csv(infile1)
df2 = pd.read_csv(infile2)
df = pd.merge(df1, df2, on="index", how="inner")

df = df[df.Mstar < 0.1]
df = df[['RAdeg', 'DEdeg', 'Mstar']]
df = df[(df.RAdeg > 83.7) & (df.RAdeg < 84)]
df = df[(df.DEdeg > -5.6) & (df.DEdeg < -5.2)]

w = WCS(header)
bx, by = w.wcs_world2pix(df.RAdeg[(df.Mstar < 0.08) & (df.Mstar > 0.0133642)],
                         df.DEdeg[(df.Mstar < 0.08) & (df.Mstar > 0.0133642)], 1)
px, py = w.wcs_world2pix(df.RAdeg[df.Mstar < 0.0133642], df.DEdeg[df.Mstar < 0.0133642], 1)
stx, sty = w.wcs_world2pix(df.RAdeg[df.Mstar >= 0.08], df.DEdeg[df.Mstar >= 0.08], 1)

bd_coords = np.vstack((bx, by)).T
planet_coords = np.vstack((px, py)).T
all_coords = np.vstack((planet_coords, bd_coords))

fov_width_arcmin = 2.2 * u.arcmin
fov_height_arcmin = 2.2 * u.arcmin
scale_x = np.abs(wcs_info.pixel_scale_matrix[0, 0]) * u.deg
scale_y = np.abs(wcs_info.pixel_scale_matrix[1, 1]) * u.deg
fov_width_pix = (fov_width_arcmin.to(u.deg) / scale_x).value
fov_height_pix = (fov_height_arcmin.to(u.deg) / scale_y).value

ny, nx = cal_image.shape
x_grid = np.arange(250, nx - fov_width_pix-200, 10)
y_grid = np.arange(150, ny - fov_height_pix-80, 10)
angles = -np.arange(250, 291, 1)

best_score = -1
best_corner = None
best_angle = None
best_counts = {}
best_sw_center_pix = None

for x0 in x_grid:
    for y0 in y_grid:
        cx = x0 + fov_width_pix / 2
        cy = y0 + fov_height_pix / 2
        for angle in angles:
            transform = Affine2D().rotate_deg_around(cx, cy, angle)

            sw_box = np.array([[x0, y0],
                               [x0 + fov_width_pix / 2, y0],
                               [x0 + fov_width_pix / 2, y0 + fov_height_pix / 2],
                               [x0, y0 + fov_height_pix / 2]])

            ne_box = np.array([[x0 + fov_width_pix / 2, y0 + fov_height_pix / 2],
                               [x0 + fov_width_pix, y0 + fov_height_pix / 2],
                               [x0 + fov_width_pix, y0 + fov_height_pix],
                               [x0 + fov_width_pix / 2, y0 + fov_height_pix]])

            nw_box = np.array([[x0, y0 + fov_height_pix / 2],
                               [x0 + fov_width_pix / 2, y0 + fov_height_pix / 2],
                               [x0 + fov_width_pix / 2, y0 + fov_height_pix],
                               [x0, y0 + fov_height_pix]])

            se_box = np.array([[x0 + fov_width_pix / 2, y0],
                               [x0 + fov_width_pix, y0],
                               [x0 + fov_width_pix, y0 + fov_height_pix / 2],
                               [x0 + fov_width_pix / 2, y0 + fov_height_pix / 2]])

            sw_path = Path(transform.transform(sw_box))
            ne_path = Path(transform.transform(ne_box))
            nw_path = Path(transform.transform(nw_box))
            se_path = Path(transform.transform(se_box))

            in_sw = sw_path.contains_points(all_coords)
            in_ne = ne_path.contains_points(all_coords)
            in_nw = nw_path.contains_points(all_coords)
            in_se = se_path.contains_points(all_coords)

            n_sw = np.sum(in_sw)
            n_ne = np.sum(in_ne)
            n_nw = np.sum(in_nw)
            n_se = np.sum(in_se)
            score = n_sw + (n_ne + n_nw + n_se)

            if score > best_score:
                best_score = score
                best_corner = (x0, y0)
                best_angle = angle
                best_sw_center_pix = transform.transform([[x0 + fov_width_pix / 4, y0 + fov_height_pix / 4]])[0]
                best_counts = {
                    'SW': n_sw, 'NE': n_ne, 'NW': n_nw, 'SE': n_se,
                    'Rest': n_ne + n_nw + n_se,
                    'Total': n_sw + n_ne + n_nw + n_se,
                    'SW_planets': np.sum(sw_path.contains_points(planet_coords)),
                    'SW_bds': np.sum(sw_path.contains_points(bd_coords)),
                    'FOV_planets': np.sum(Path(transform.transform([[x0, y0], [x0 + fov_width_pix, y0],
                                                                      [x0 + fov_width_pix, y0 + fov_height_pix],
                                                                      [x0, y0 + fov_height_pix]])).contains_points(planet_coords)),
                    'FOV_bds': np.sum(Path(transform.transform([[x0, y0], [x0 + fov_width_pix, y0],
                                                                [x0 + fov_width_pix, y0 + fov_height_pix],
                                                                [x0, y0 + fov_height_pix]])).contains_points(bd_coords))
                }

corner_x, corner_y = best_corner
cx = corner_x + fov_width_pix / 2
cy = corner_y + fov_height_pix / 2
transform = Affine2D().rotate_deg_around(cx, cy, best_angle)
sw_center_ra, sw_center_dec = w.wcs_pix2world(best_sw_center_pix[0], best_sw_center_pix[1], 1)

Gaia.ROW_LIMIT = 20000
coord = SkyCoord(ra=sw_center_ra, dec=sw_center_dec, unit="deg")
job = Gaia.cone_search_async(coord, radius=u.Quantity(1.5, u.deg))
gaia_results = job.get_results()

# Gaia coords in pixel
gaia_coords = np.column_stack((gaia_results['ra'], gaia_results['dec']))
gaia_pix = wcs_info.wcs_world2pix(gaia_coords, 1)
gaia_x = gaia_pix[:, 0]
gaia_y = gaia_pix[:, 1]

theta_opt = np.deg2rad(-best_angle)
rot_matrix = np.array([[np.cos(theta_opt), -np.sin(theta_opt)],
                       [np.sin(theta_opt), np.cos(theta_opt)]])
gaia_points = np.vstack((gaia_x - cx, gaia_y - cy))
gaia_rot = rot_matrix @ gaia_points
in_fov = (np.abs(gaia_rot[0]) < fov_width_pix / 2) & (np.abs(gaia_rot[1]) < fov_height_pix / 2)
gaia_in_fov_count = np.sum(in_fov)

gaia_onc_mask = (gaia_results['parallax'] > 2.22) & (gaia_results['parallax'] < 2.86)
gaia_onc = gaia_results[gaia_onc_mask]
onc_coords = np.column_stack((gaia_onc['ra'], gaia_onc['dec']))
onc_pix = wcs_info.wcs_world2pix(onc_coords, 1)
onc_x = onc_pix[:, 0]
onc_y = onc_pix[:, 1]
onc_points = np.vstack((onc_x - cx, onc_y - cy))
onc_rot = rot_matrix @ onc_points
in_fov_onc = (np.abs(onc_rot[0]) < fov_width_pix / 2) & (np.abs(onc_rot[1]) < fov_height_pix / 2)
gaia_onc_in_fov_count = np.sum(in_fov_onc)

gaia_mags = gaia_results['phot_g_mean_mag']
gaia_marker_size = -4.5 * gaia_mags + 90
gaia_marker_size[gaia_marker_size < 5] = 5

# --- Final plot ---
plt.figure(figsize=(11, 11))
ax = plt.subplot(111, projection=wcs_info)
vmin, vmax = np.nanpercentile(cal_image, [1.0, 99.9])
norm = ImageNormalize(vmin=vmin, vmax=vmax, stretch=AsinhStretch())
plt.imshow(cal_image, origin='lower', cmap='gray', norm=norm)

plt.scatter(bx, by, edgecolors='magenta', facecolors='none', s=45, label='HST BDs',zorder=10)
plt.scatter(px, py, edgecolors='yellow', facecolors='none', s=50, label='HST Planets',zorder=11)
plt.scatter(gaia_x[in_fov], gaia_y[in_fov],
            marker='s', c='cyan', s=gaia_marker_size[in_fov],
            zorder=2, edgecolors='black', lw=0.3, label='Gaia targets')
plt.scatter(stx, sty, edgecolors='black', facecolors='none', s=55, zorder=9,
            marker='*', linewidth=1.0, label='HST Stars')

fov_rect = Rectangle((corner_x, corner_y), fov_width_pix, fov_height_pix,
                     edgecolor='white', facecolor='none', linewidth=1,ls='dashed',
                     transform=transform + ax.transData, label='Best JWST NIRCam FOV')
sw_rect = Rectangle((corner_x, corner_y), fov_width_pix / 2, fov_height_pix / 2,
                    edgecolor='red', facecolor='none', linewidth=2.5, linestyle='dotted',
                    transform=transform + ax.transData, label='SW Quadrant')
ax.add_patch(fov_rect)
ax.add_patch(sw_rect)
plt.scatter(best_sw_center_pix[0], best_sw_center_pix[1], s=40, color='red', marker='x', zorder=20, label='SW Center')

plt.title(
    f'Optimal FOV | SW Center RA/Dec: ({sw_center_ra:.5f}, {sw_center_dec:.5f}), Angle: {abs(best_angle)}° CW\n'
    f'Total: {best_counts["Total"]} | Planets: {best_counts["FOV_planets"]} | BDs: {best_counts["FOV_bds"]}\n'
    f'SW Substellar Counts: {best_counts["SW"]} | Planets: {best_counts["SW_planets"]} | BDs: {best_counts["SW_bds"]}\n'
    f'Outside SW Substellar Counts: {best_counts["Rest"]}\n'
    f'Gaia targets in FOV: {gaia_in_fov_count}\n'
    f'Gaia ONC targets (350–450 pc) in FOV: {gaia_onc_in_fov_count}'
)

plt.xlabel('RA')
plt.ylabel('DEC')
plt.legend(frameon=True, facecolor='white', loc='upper left')
plt.tight_layout()
plt.ylim(200, 600)
plt.xlim(300, 800)

plt.savefig('/Users/melindasoares/Desktop/jwst_onc_optimal_quadrant_weighted_with_gaia_overlay.png', dpi=300, facecolor='white')
plt.show()
