from SEDFit.sed import SEDFit
import matplotlib.pyplot as plt
x=SEDFit('07:45:28.973','-52:22:59.73',5,grid_type='phoenix')
x.addguesses(dist=406.,av=0.7,r=[0.5,1.0],teff=[3660,5800],logg=4.4)
x.addrange(dist=[400,420],r=[0.4,1.2],logg=[3.7,4.4],feh=[-0.7,0.5],teff=[3000,6000])
idx=range(1,len(x.sed))
pars=x.fit(use_gaia=True,use_mag=idx)

print("Radius: {} Rsun".format(x.getr()))
print("Teff: {} K".format(x.getteff()))
