def makeplot(name,gaia_logg,gaia_teff,tres_logg,tres_teff,tres_snr,sed_logg,sed_teff,loc='lower right',color='red'):
    tresteff_weighted_average = np.average(tres_teff, weights=tres_snr)
    treslogg_weighted_average = np.average(tres_logg, weights=tres_snr)
    plt.scatter(treslogg_weighted_average,tresteff_weighted_average,s=100,marker='*',
                edgecolors='black',color=color)
    plt.scatter(tres_logg,tres_teff,s=40,edgecolors='black',color=color)
    plt.scatter(gaia_logg,gaia_teff,s=40,edgecolors='black',color=color)
    plt.scatter(sed_logg,sed_teff,s=40,edgecolors='black',color=color)
    return
